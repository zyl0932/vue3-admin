# 5.Vue3 组件库配置

## 5-1.按需导入

将全局引入`element-plus`方式改成按需导入的方式
<br>

首先你需要安装`unplugin-vue-components`和`unplugin-auto-import`这两款插件

```shell
pnpm install -D unplugin-vue-components unplugin-auto-import
```

```ts
import { defineConfig } from "vite";
import vue from "@vitejs/plugin-vue";
import path from "path";
import { createSvgIconsPlugin } from "vite-plugin-svg-icons";
// 自动导入
import AutoImport from "unplugin-auto-import/vite";
import Components from "unplugin-vue-components/vite";
import { ElementPlusResolver } from "unplugin-vue-components/resolvers";
export default defineConfig({
  resolve: {
    alias: [
      {
        find: "@",
        replacement: path.resolve(__dirname, "src"),
      },
    ],
  },
  plugins: [
    vue(),
    createSvgIconsPlugin({
      iconDirs: [path.resolve(process.cwd(), "src/icons/svg")],
      symbolId: "icon-[name]",
      inject: "body-last",
      customDomId: "__svg__icons__dom__",
    }),
    // 自动导入
    AutoImport({
      imports: ["vue", "vue-router"], // 顺便自动导入vue vue-router
      resolvers: [ElementPlusResolver()],
      dts: "src/auto-import.d.ts", // 生成的全局变量放到此目录下
    }),
    Components({
      // 默认只针对src/components目录实现自动导入
      dirs: ["src/components", "src/layout/components"], // 后面布局组件也有相关的组件期望自动导入
      dts: "src/components.d.ts",
      resolvers: [ElementPlusResolver()], // 生成的组件的类型放到这里
    }),
  ],
});
```

## 5-2element-plus 组件 api 挂载到 app

### 添加文件`src/plugins/element.ts`

```ts
import { App } from "vue";
import { ElMessage, ElNotification, ElMessageBox } from "element-plus";

// 默认自动导入组件插件，只会解析在模板中使用的组件,所以这里需要导入样式
// 这里的样式也可以做到按需导入
import "element-plus/theme-chalk/el-message.css";
import "elemen-plus/theme-chalk/el-notification.css";
import "element-plus/theme-chalk/el-message-box.css";
// 按需导入组件列表
// Vue.prototype 替换为 config.globalProperties
// 文档说明 https://cn.vuejs.org/#vue-prototype-%E6%9B%BF%E6%8D%A2%E4%B8%BA-config-globalproperties
app.config.globalProperties.$message = ElMessage;
app.config.globalProperties.$notify = ElNotification;
app.config.globalProperties.$confirm = ElMessageBox.confirm;
app.config.globalProperties.$alert = ElMessage.alert;
app.config.globalProperties.$prompt = ElMessageBox.prompt;
```

### src/main.ts

```ts
import { createApp } from "vue";
import App from "./App.vue";
import router from "./router/index";
import { createPinia} "pinia"
import "normalize.css/normalize.css"
import "@/styles/index.scss"

import initSvgIcon from "@/icons/index"
import "vitual:svg-icons-register"

// 注册element-plus
import installElementPlus from "./plugins/element"
const app = createApp(app)
app.use(createPinia())
// 安装element-plus插件
app.use(installElementPlus)
app.use(router)
app.use(initSvgIcon)
app.mount("#app")
```

// 这里要注意的是项目需要使用其他`element-plus`组件，这样自动导入插件就会默认引入公共样式，否则公共样式无法导入,弹框就没有颜色了!

## 5-3 按需导入样式

如果不想手动导入样式，可以通过`unplugin-element-plus`自动解析导入

> 安装`pnpm install unplugin-element-plus`

```ts
import { defineConfig } from "vite";
import vue from "@vitejs/plugin-vue";
import path from "path";
import { createSvgIconsPlugin } from "vite-plugin-svg-icons";
// 自动导入
import AutoImport from "unplugin-auto-import/vite";
import Components from "unplugin-vue-components/vite";
import { ElementPlusResolver } from "unplugin-vue-components/resolvers";
import {
  createStyleImportPlugin,
  ElementPlusResolve,
} from "vite-plugin-style-import";

export default defineConfig({
  resolve: {
    alias: [
      {
        find: "@",
        replacement: path.resolve(__dirname, "src"),
      },
    ],
  },
  plugins: [
    ...,
    ElementPlus()
  ]
});
```

去掉`plugins/element.ts`中的引入即可

```ts

```

## 5-4.组件中使用

### `src/views/dashboard/index.vue`

```vue
<template>
  <div>
    <h1>Dashboard page</h1>
    <svg-icon icon-class="bug"></svg-icon>
    <svg-icon
      icon-class="404"
      class-name="custom-class"
      @click="sayHi"
    ></svg-icon>
    <br />
    <br />
    <br />
    <el-button>按钮</el-button>
  </div>
</template>
<script setup lang="ts">
// import { getCurrentInstance } from "vue"
const { proxy } = getCurrentInstance()!;
const sayHi = () => {
  proxy?.$message.success("恭喜你，这是一条成功的消息");
};
</script>
<style lang="scss">
.custom-class {
  // 自定义样式404
  font-size: 200px;
  color: green;
}
</style>
```

## 5-5 解决 eslint 报错问题

当我们去掉 Vue 导入到变量后，虽然可以正常运行,但是`eslint`提示出错

```ts
AutoImport({
  imports: ["vue", "vue-router"],
  resolvers: [ElementPlusResolver],
  eslint: [enabled: false], // 改成true生成一次后禁用即可
  dts: "src/auto-import.d.ts"
});
```

在.eslintrc 文件中引入生成的配置

```ts
module.exports = {
  extends: [
    "eslint:recommended",
    "plugin:vue/vue3-essential",
    "plugin:@typescript-eslint/recommended",
    "@vue/prettier",
    "./eslint-auto-import.json",
  ],
  plugins: ["vue", "@typescript-eslint"],
  rules: {
    "vue/multi-word-component-names": "off",
    "prettier/prettier": [
      "error",
      {
        singleQuote: false, // 使用单引号
        semi: false, // 末尾添加分号
        tabWidthComma: "none",
        useTabs: false,
        endOfLine: "auto",
      },
    ],
    "@typescript-eslint/no-non-null-assertion": "
    off"
  },
};
```
